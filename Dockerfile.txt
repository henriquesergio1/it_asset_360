# Stage 1: Build the React Application
FROM node:20-alpine as build

WORKDIR /app

# Install dependencies separately for cache
COPY package.json ./
RUN npm install

# Copy all source
COPY . .

# --- PRE-BUILD PREPARATION (v3.5.6) ---
# Ensure clean workspace and CRA structure
RUN rm -rf src public build
RUN mkdir -p src public

# Handle entry points at root if they exist
RUN [ -f index.tsx ] && mv index.tsx src/ || true
RUN [ -f App.tsx ] && mv App.tsx src/ || true
RUN [ -f types.ts ] && mv types.ts src/ || true
RUN [ -f index.html ] && mv index.html public/ || true
RUN [ -f metadata.json ] && mv metadata.json public/ || true

# Move core logical folders
RUN mv components contexts services utils src/ 2>/dev/null || true

# Move any other TSX files left at root to src
RUN find . -maxdepth 1 -name "*.tsx" -o -name "*.ts" ! -name "server.js" -exec mv {} src/ \; 2>/dev/null || true

# Build configuration
ENV CI=false
ENV TSC_COMPILE_ON_ERROR=true
ENV ESLINT_NO_DEV_ERRORS=true
ENV GENERATE_SOURCEMAP=false

RUN npm run build

# Stage 2: Serve with Nginx
FROM nginx:alpine

# Clear default static files
RUN rm -rf /usr/share/nginx/html/*

# Copy build output
COPY --from=build /app/build /usr/share/nginx/html

# Robust permissions fix
RUN chmod -R 755 /usr/share/nginx/html && \
    chown -R nginx:nginx /usr/share/nginx/html

# Custom nginx config
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
