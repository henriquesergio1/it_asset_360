# Stage 1: Build the React Application
FROM node:20-alpine as build

WORKDIR /app

# Copy package.json and install dependencies
COPY package.json ./
RUN npm install

# Copy source code
COPY . .

# --- FIX STRUCTURE FOR REACT SCRIPTS (v3.5.3) ---
# Clean redundant files that cause build conflicts
RUN rm -f index.prod.tsx SystemInfoModal.tsx 2>/dev/null || true

# Create standard CRA directories
RUN mkdir -p public src

# Move Static Assets to public/
RUN [ -f index.html ] && mv index.html public/
RUN [ -f metadata.json ] && mv metadata.json public/

# Move Source Code to src/
# We move components first, then root level code
RUN mv components contexts services utils src/ 2>/dev/null || true
RUN mv *.tsx *.ts src/ 2>/dev/null || true

# IMPORTANT: Set CI=false and skip sourcemaps to optimize build and treat warnings as non-fatal
ENV CI=false
ENV GENERATE_SOURCEMAP=false
RUN npm run build

# Stage 2: Serve with Nginx
FROM nginx:alpine

# Remove default nginx static assets
RUN rm -rf /usr/share/nginx/html/*

# Copy static assets from builder stage
COPY --from=build /app/build /usr/share/nginx/html

# Copy custom nginx config
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Expose port 80
EXPOSE 80

# Start Nginx
CMD ["nginx", "-g", "daemon off;"]
