# Stage 1: Build the React Application
FROM node:20-alpine as build

WORKDIR /app

# Copy package.json and install dependencies
COPY package.json ./
RUN npm install

# Copy source code
COPY . .

# --- BUILD RESILIENCE (v3.5.4) ---
# 1. Remove redundant entry points and duplicate components at root level
RUN rm -f index.prod.tsx SystemInfoModal.tsx 2>/dev/null || true

# 2. Setup standard CRA directory structure
RUN mkdir -p public src

# 3. Move static assets to public
RUN [ -f index.html ] && mv index.html public/
RUN [ -f metadata.json ] && mv metadata.json public/

# 4. Move logical folders into src
RUN mv components contexts services utils src/ 2>/dev/null || true

# 5. Move root logic files into src
RUN mv *.tsx *.ts src/ 2>/dev/null || true

# --- EXECUTE BUILD ---
# CI=false: Warnings won't break the build
# TSC_COMPILE_ON_ERROR: Allows build to complete even with minor TS errors
# ESLINT_NO_DEV_ERRORS: Ignores linting errors during build
# GENERATE_SOURCEMAP=false: Saves memory during compilation
ENV CI=false
ENV TSC_COMPILE_ON_ERROR=true
ENV ESLINT_NO_DEV_ERRORS=true
ENV GENERATE_SOURCEMAP=false

RUN npm run build

# Stage 2: Serve with Nginx
FROM nginx:alpine

# Remove default assets
RUN rm -rf /usr/share/nginx/html/*

# Copy built app
COPY --from=build /app/build /usr/share/nginx/html

# Copy custom nginx config
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
